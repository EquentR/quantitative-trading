# Backend HTTP API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a FastAPI-backed HTTP API and unified background service for the existing manual ledger, cash account, account snapshot, and scheduler controls.

**Architecture:** Add a thin API layer over existing services and repositories, with SQLite-backed auth state, scheduler state, and account snapshot retrieval. `qt service run` becomes the unified foreground service command that starts the API server and restores the persisted scheduler state.

**Tech Stack:** Python 3.11+, FastAPI, Uvicorn, Pydantic v2, SQLite, Typer, APScheduler, pytest, FastAPI TestClient.

---

## Scope And Guardrails

This plan implements `docs/superpowers/specs/2026-07-07-http-api-design.md`.

Keep these constraints visible during every task:

- Do not implement automatic real trading.
- Do not click or control any real trading client.
- Do not read, store, log, or commit real broker credentials, cookies, API keys, tokens, or passwords.
- API write operations only maintain the manual ledger and manual cash account.
- API, CLI, and scheduler must share the same service and repository logic.
- Use fake market providers in tests; do not make core tests depend on live AkShare network calls.
- Keep API response fields in `snake_case`.
- Convert API errors to `{ "error": { "code", "message", "details" } }`.

```python
from __future__ import annotations

import secrets
import sqlite3
from dataclasses import dataclass
from datetime import UTC, datetime


@dataclass(frozen=True)
class ApiAuthState:
    password_hash: str | None
    token_secret: str
    updated_at: datetime

    @property
    def is_configured(self) -> bool:
        return self.password_hash is not None


class ApiAuthRepository:
    def __init__(self, connection: sqlite3.Connection) -> None:
        self.connection = connection

    def get(self) -> ApiAuthState:
        row = self.connection.execute(
            """
            SELECT password_hash, token_secret, updated_at
            FROM api_auth_state
            WHERE id = 1
            """
        ).fetchone()
        if row is not None:
            return self._from_row(row)

        now = datetime.now(UTC)
        token_secret = secrets.token_urlsafe(32)
        with self.connection:
            self.connection.execute(
                """
                INSERT INTO api_auth_state (id, password_hash, token_secret, updated_at)
                VALUES (1, NULL, ?, ?)
                """,
                (token_secret, now.isoformat()),
            )
        return ApiAuthState(
            password_hash=None,
            token_secret=token_secret,
            updated_at=now,
        )

    def save_password_hash(self, password_hash: str, *, now: datetime) -> ApiAuthState:
        state = self.get()
        with self.connection:
            self.connection.execute(
                """
                UPDATE api_auth_state
                SET password_hash = ?, updated_at = ?
                WHERE id = 1
                """,
                (password_hash, now.isoformat()),
            )
        return ApiAuthState(
            password_hash=password_hash,
            token_secret=state.token_secret,
            updated_at=now,
        )

    def save_token_secret(self, token_secret: str, *, now: datetime) -> ApiAuthState:
        state = self.get()
        with self.connection:
            self.connection.execute(
                """
                UPDATE api_auth_state
                SET token_secret = ?, updated_at = ?
                WHERE id = 1
                """,
                (token_secret, now.isoformat()),
            )
        return ApiAuthState(
            password_hash=state.password_hash,
            token_secret=token_secret,
            updated_at=now,
        )

    @staticmethod
    def _from_row(row: sqlite3.Row) -> ApiAuthState:
        return ApiAuthState(
            password_hash=row["password_hash"],
            token_secret=row["token_secret"],
            updated_at=datetime.fromisoformat(row["updated_at"]),
        )
```

from __future__ import annotations

import sqlite3
from dataclasses import dataclass
from datetime import datetime


@dataclass(frozen=True)
class SchedulerState:
    enabled: bool
    interval_seconds: int
    run_on_start: bool
    last_started_at: datetime | None
    last_finished_at: datetime | None
    last_status: str | None
    last_reason: str | None
    last_error: str | None
    last_snapshot_id: int | None
    updated_at: datetime


class SchedulerStateRepository:
    def __init__(self, connection: sqlite3.Connection) -> None:
        self.connection = connection

    def get_or_create(
        self,
        *,
        interval_seconds: int,
        run_on_start: bool,
        now: datetime,
    ) -> SchedulerState:
        row = self.connection.execute(
            """
            SELECT *
            FROM scheduler_state
            WHERE id = 1
            """
        ).fetchone()
        if row is not None:
            return self._from_row(row)

        with self.connection:
            self.connection.execute(
                """
                INSERT INTO scheduler_state (
                  id,
                  enabled,
                  interval_seconds,
                  run_on_start,
                  updated_at
                ) VALUES (
                  1,
                  0,
                  ?,
                  ?,
                  ?
                )
                """,
                (interval_seconds, int(run_on_start), now.isoformat()),
            )
        return SchedulerState(
            enabled=False,
            interval_seconds=interval_seconds,
            run_on_start=run_on_start,
            last_started_at=None,
            last_finished_at=None,
            last_status=None,
            last_reason=None,
            last_error=None,
            last_snapshot_id=None,
            updated_at=now,
        )

    def set_enabled(
        self,
        enabled: bool,
        *,
        interval_seconds: int,
        run_on_start: bool,
        now: datetime,
    ) -> SchedulerState:
        self.get_or_create(
            interval_seconds=interval_seconds,
            run_on_start=run_on_start,
            now=now,
        )
        with self.connection:
            self.connection.execute(
                """
                UPDATE scheduler_state
                SET
                  enabled = ?,
                  interval_seconds = ?,
                  run_on_start = ?,
                  updated_at = ?
                WHERE id = 1
                """,
                (int(enabled), interval_seconds, int(run_on_start), now.isoformat()),
            )
        return self.get_or_create(
            interval_seconds=interval_seconds,
            run_on_start=run_on_start,
            now=now,
        )

    def record_result(
        self,
        *,
        started_at: datetime,
        finished_at: datetime,
        status: str,
        reason: str,
        error: str | None,
        snapshot_id: int | None,
        now: datetime,
    ) -> SchedulerState:
        current = self.get_or_create(
            interval_seconds=1,
            run_on_start=True,
            now=now,
        )
        with self.connection:
            self.connection.execute(
                """
                UPDATE scheduler_state
                SET
                  last_started_at = ?,
                  last_finished_at = ?,
                  last_status = ?,
                  last_reason = ?,
                  last_error = ?,
                  last_snapshot_id = ?,
                  updated_at = ?
                WHERE id = 1
                """,
                (
                    started_at.isoformat(),
                    finished_at.isoformat(),
                    status,
                    reason,
                    error,
                    snapshot_id,
                    now.isoformat(),
                ),
            )
        return self.get_or_create(
            interval_seconds=current.interval_seconds,
            run_on_start=current.run_on_start,
            now=now,
        )

    @staticmethod
    def _from_row(row: sqlite3.Row) -> SchedulerState:
        return SchedulerState(
            enabled=bool(row["enabled"]),
            interval_seconds=row["interval_seconds"],
            run_on_start=bool(row["run_on_start"]),
            last_started_at=_parse_datetime(row["last_started_at"]),
            last_finished_at=_parse_datetime(row["last_finished_at"]),
            last_status=row["last_status"],
            last_reason=row["last_reason"],
            last_error=row["last_error"],
            last_snapshot_id=row["last_snapshot_id"],
            updated_at=datetime.fromisoformat(row["updated_at"]),
        )


def _parse_datetime(value: str | None) -> datetime | None:
    if value is None:
        return None
    return datetime.fromisoformat(value)
```

from typing import Any

from fastapi import FastAPI, Request
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse


class ApiError(Exception):
    def __init__(
        self,
        *,
        status_code: int,
        code: str,
        message: str,
        details: dict[str, Any] | None = None,
    ) -> None:
        self.status_code = status_code
        self.code = code
        self.message = message
        self.details = details or {}


def error_body(*, code: str, message: str, details: dict[str, Any] | None = None) -> dict[str, Any]:
    return {
        "error": {
            "code": code,
            "message": message,
            "details": details or {},
        }
    }


def install_error_handlers(app: FastAPI) -> None:
    @app.exception_handler(ApiError)
    async def handle_api_error(request: Request, exc: ApiError) -> JSONResponse:
        return JSONResponse(
            status_code=exc.status_code,
            content=error_body(
                code=exc.code,
                message=exc.message,
                details=exc.details,
            ),
        )

    @app.exception_handler(RequestValidationError)
    async def handle_validation_error(
        request: Request,
        exc: RequestValidationError,
    ) -> JSONResponse:
        return JSONResponse(
            status_code=422,
            content=error_body(
                code="validation_error",
                message="request validation failed",
                details={"errors": exc.errors()},
            ),
        )

from __future__ import annotations

import base64
import hashlib
import hmac
import json
import secrets
from dataclasses import dataclass
from datetime import UTC, datetime, timedelta
from typing import Any

from quantitative_trading.storage.api_auth import ApiAuthRepository


class AuthSetupRequiredError(ValueError):
    pass


class AuthAlreadyConfiguredError(ValueError):
    pass


class InvalidCredentialsError(ValueError):
    pass


class InvalidTokenError(ValueError):
    pass


@dataclass(frozen=True)
class LoginResult:
    access_token: str
    token_type: str
    expires_at: datetime


@dataclass(frozen=True)
class TokenClaims:
    user: str
    expires_at: datetime


class AuthService:
    def __init__(
        self,
        repository: ApiAuthRepository,
        *,
        token_ttl_seconds: int,
        startup_password: str | None = None,
        configured_token_secret: str | None = None,
    ) -> None:
        self._repository = repository
        self._token_ttl_seconds = token_ttl_seconds
        self._startup_password = startup_password
        self._configured_token_secret = configured_token_secret
        if configured_token_secret:
            self._repository.save_token_secret(
                configured_token_secret,
                now=datetime.now(UTC),
            )

    def status(self) -> str:
        state = self._repository.get()
        if state.password_hash is not None or self._startup_password:
            return "configured"
        return "setup_required"

    def setup_password(self, password: str, *, now: datetime | None = None) -> None:
        state = self._repository.get()
        if state.password_hash is not None:
            raise AuthAlreadyConfiguredError("api password already configured")
        self._repository.save_password_hash(
            _hash_password(password),
            now=now or datetime.now(UTC),
        )

    def login(self, password: str, *, now: datetime | None = None) -> LoginResult:
        issued_at = now or datetime.now(UTC)
        state = self._repository.get()
        if state.password_hash is None and not self._startup_password:
            raise AuthSetupRequiredError("api password setup required")

        if state.password_hash is not None:
            valid = _verify_password(password, state.password_hash)
        else:
            valid = hmac.compare_digest(password, self._startup_password or "")
        if not valid:
            raise InvalidCredentialsError("invalid password")

        expires_at = issued_at + timedelta(seconds=self._token_ttl_seconds)
        token = _encode_token(
            {"user": "local", "exp": expires_at.isoformat()},
            secret=state.token_secret,
        )
        return LoginResult(
            access_token=token,
            token_type="bearer",
            expires_at=expires_at,
        )

    def verify_token(self, token: str, *, now: datetime | None = None) -> TokenClaims:
        checked_at = now or datetime.now(UTC)
        state = self._repository.get()
        payload = _decode_token(token, secret=state.token_secret)
        expires_at = datetime.fromisoformat(str(payload["exp"]))
        if expires_at <= checked_at:
            raise InvalidTokenError("token expired")
        return TokenClaims(user=str(payload["user"]), expires_at=expires_at)


def _hash_password(password: str) -> str:
    salt = secrets.token_hex(16)
    digest = hashlib.pbkdf2_hmac(
        "sha256",
        password.encode("utf-8"),
        salt.encode("utf-8"),
        100_000,
    ).hex()
    return f"pbkdf2_sha256$100000${salt}${digest}"


def _verify_password(password: str, password_hash: str) -> bool:
    algorithm, iterations, salt, expected = password_hash.split("$", 3)
    if algorithm != "pbkdf2_sha256":
        return False
    digest = hashlib.pbkdf2_hmac(
        "sha256",
        password.encode("utf-8"),
        salt.encode("utf-8"),
        int(iterations),
    ).hex()
    return hmac.compare_digest(digest, expected)


def _encode_token(payload: dict[str, Any], *, secret: str) -> str:
    payload_json = json.dumps(payload, separators=(",", ":"), sort_keys=True).encode("utf-8")
    payload_part = base64.urlsafe_b64encode(payload_json).decode("ascii").rstrip("=")
    signature = hmac.new(secret.encode("utf-8"), payload_part.encode("ascii"), hashlib.sha256)
    signature_part = base64.urlsafe_b64encode(signature.digest()).decode("ascii").rstrip("=")
    return f"{payload_part}.{signature_part}"


def _decode_token(token: str, *, secret: str) -> dict[str, Any]:
    try:
        payload_part, signature_part = token.split(".", 1)
    except ValueError as exc:
        raise InvalidTokenError("invalid token") from exc
    expected = hmac.new(secret.encode("utf-8"), payload_part.encode("ascii"), hashlib.sha256)
    expected_part = base64.urlsafe_b64encode(expected.digest()).decode("ascii").rstrip("=")
    if not hmac.compare_digest(signature_part, expected_part):
        raise InvalidTokenError("invalid token")
    padded = payload_part + "=" * (-len(payload_part) % 4)
    return json.loads(base64.urlsafe_b64decode(padded.encode("ascii")))
```

    authorization: str | None = Header(default=None, alias="Authorization"),
    container: ApiContainer = Depends(get_container),
) -> TokenClaims:

@dataclass(frozen=True)
class CreatedSnapshot:
    snapshot_id: int
    snapshot: AccountSnapshot


def market_provider_from_settings(settings) -> MarketDataProvider:
    if not settings.enable_market_fetch:
        return DisabledMarketProvider()
    if settings.market_provider.strip().lower() == "akshare":
        return AkShareMarketProvider()
    raise ApiError(
        status_code=422,
        code="validation_error",
        message=f"unsupported market provider: {settings.market_provider}",
    )


def create_and_save_account_snapshot(settings: Settings) -> CreatedSnapshot:
    with connect(settings) as connection:
        migrate(connection)
        ledger = ReadOnlyLedgerService(PositionRepository(connection))
        cash = ReadOnlyCashService(CashAccountRepository(connection))
        service = AccountService(
            ledger=ledger,
            cash=cash,
            market=market_provider_from_settings(settings),
        )
        snapshot = service.create_snapshot()
        cash_account = cash.get_account()
        positions = ledger.list_positions()
        ledger_max_updated_at = max(
            (position.updated_at for position in positions),
            default=None,
        )
        snapshot_id = AccountSnapshotRepository(connection).save(
            snapshot,
            cash_account_updated_at=cash_account.updated_at if cash_account else None,
            ledger_max_updated_at=ledger_max_updated_at,
        )
    return CreatedSnapshot(snapshot_id=snapshot_id, snapshot=snapshot)

        job: Callable[[str], object],
        scheduler_factory: Callable[..., Any] = BackgroundScheduler,
    ) -> None:
        self._interval_seconds = interval_seconds
        self._timezone = timezone
        self._job = job
        self._scheduler_factory = scheduler_factory
        self._scheduler = None

    @property
    def is_running(self) -> bool:
        return self._scheduler is not None

    def start(self) -> bool:
        if self._scheduler is not None:
            return False
        scheduler = self._scheduler_factory(timezone=self._timezone)
        scheduler.add_job(
            lambda: self._job("intraday"),
            trigger="interval",
            seconds=self._interval_seconds,
            id="account_snapshot_intraday",
            max_instances=1,
            replace_existing=True,
        )
        scheduler.start()
        self._scheduler = scheduler
        return True

    def stop(self) -> bool:
        if self._scheduler is None:
            return False
        self._scheduler.shutdown(wait=False)
        self._scheduler = None
        return True
```

from __future__ import annotations

from datetime import UTC, datetime

import uvicorn

from quantitative_trading.api.app import create_app
from quantitative_trading.config import Settings
from quantitative_trading.runtime.account_snapshot_job import create_and_save_account_snapshot
from quantitative_trading.runtime.scheduler import SchedulerManager
from quantitative_trading.storage.scheduler_state import SchedulerStateRepository
from quantitative_trading.storage.sqlite import connect, migrate


def run_api_service(settings: Settings) -> None:
    def run_snapshot_job(reason: str) -> None:
        started_at = datetime.now(UTC)
        status = "success"
        error = None
        snapshot_id = None
        try:
            created = create_and_save_account_snapshot(settings)
            snapshot_id = created.snapshot_id
        except Exception as exc:
            status = "failed"
            error = str(exc)
        finished_at = datetime.now(UTC)
        with connect(settings) as connection:
            migrate(connection)
            SchedulerStateRepository(connection).record_result(
                started_at=started_at,
                finished_at=finished_at,
                status=status,
                reason=reason,
                error=error,
                snapshot_id=snapshot_id,
                now=finished_at,
            )

    scheduler = SchedulerManager(
        interval_seconds=settings.intraday_interval_seconds,
        timezone=settings.timezone,
        job=run_snapshot_job,
    )
    app = create_app(settings, scheduler=scheduler, restore_scheduler=True)
    with connect(settings) as connection:
        migrate(connection)
        state = SchedulerStateRepository(connection).get_or_create(
            interval_seconds=settings.intraday_interval_seconds,
            run_on_start=settings.service_run_on_start_when_scheduler_enabled,
            now=datetime.now(UTC),
        )
    if state.enabled and state.run_on_start:
        run_snapshot_job("startup")
    uvicorn.run(app, host=settings.api_host, port=settings.api_port)
```

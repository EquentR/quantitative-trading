# Roadmap P1-P4 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the post-P0 trading decision workflow from stock universe and cached inputs through plans, recommendations, audit, notifications, feedback, and Web presentation.

**Architecture:** Implement the backend first as shared Python services and repositories, then expose the same services through CLI, FastAPI, scheduler jobs, and Web queries. Preserve adapter boundaries: external data enters only through market/watchlist adapters and is converted to internal Pydantic models before strategy, risk, or recommendation logic sees it. Web remains a local console that displays and maintains backend state; it does not compute signals or recommendations.

**Tech Stack:** Python 3.11+, Pydantic v2, SQLite, Typer CLI, FastAPI, APScheduler, pytest, Vue 3, TypeScript, TanStack Query for Vue, Pinia, Vitest, Testing Library, Playwright.

## Safety Rules

- Do not implement automatic real trading.
- Do not control or click a real trading client.
- Do not read, save, log, or commit real broker credentials.
- Do not derive real holdings or cash from Eastmoney simulation or market data.
- Manual position ledger remains the authority for holdings, cost, quantity, and available quantity.
- Manual cash account remains the authority for cash balance, net principal, and buying cash.
- Strategy and risk modules must not import AkShare, requests, Eastmoney skill files, or raw third-party response shapes.
- Recommendation output must include reason, risk, invalidation conditions, position constraints, and data timestamps.

```python
from __future__ import annotations

from datetime import datetime
from enum import StrEnum
from typing import Any

from pydantic import BaseModel, ConfigDict, Field, field_validator


class WatchPinnedSource(StrEnum):
    MANUAL = "manual"
    SYNCED = "synced"
    MANUAL_SYNCED = "manual_synced"


class WatchPinnedInput(BaseModel):
    model_config = ConfigDict(str_strip_whitespace=True)

    symbol: str = Field(pattern=r"^\d{6}$")
    name: str = Field(min_length=1)
    rank: int = Field(ge=1)
    plan_enabled: bool = False
    note: str = ""


class WatchPinnedItem(WatchPinnedInput):
    source: WatchPinnedSource
    updated_at: datetime

    @field_validator("updated_at")
    @classmethod
    def updated_at_must_be_timezone_aware(cls, value: datetime, info: Any) -> datetime:
        if value.tzinfo is None or value.utcoffset() is None:
            raise ValueError(f"{info.field_name} must be timezone-aware")
        return value
```

```python
class WatchPinnedRepository:
    def __init__(self, connection: sqlite3.Connection) -> None: ...
    def list(self) -> list[WatchPinnedItem]: ...
    def get(self, symbol: str) -> WatchPinnedItem | None: ...
    def upsert(self, item: WatchPinnedInput, *, source: WatchPinnedSource, now: datetime) -> WatchPinnedItem: ...
    def remove(self, symbol: str) -> None: ...
    def replace_all(self, items: list[WatchPinnedInput], *, source: WatchPinnedSource, now: datetime) -> list[WatchPinnedItem]: ...
    def merge_synced(self, items: list[WatchPinnedInput], *, now: datetime) -> list[WatchPinnedItem]: ...
```

Implementation rules:

- Validate all `WatchPinnedInput` values before writing.
- Duplicate symbols in `replace_all` raise `ValueError`.
- `merge_synced` preserves local `plan_enabled` and `note` for existing symbols.
- `merge_synced` updates `name` and `rank` from synced rows.
- New synced rows are written with `plan_enabled=False`, `note=""`, and `source="synced"`.
- Existing local rows not present in sync remain with `source="manual"`.
- Use one SQLite transaction per bulk operation.

```python
from __future__ import annotations

from datetime import datetime

from quantitative_trading.ledger.models import Position
from quantitative_trading.universe.models import UniverseMember, UniverseSource
from quantitative_trading.watchlist.models import WatchPinnedItem


def build_universe(
    *,
    positions: list[Position],
    watchlist: list[WatchPinnedItem],
    created_at: datetime,
) -> list[UniverseMember]:
    by_symbol: dict[str, UniverseMember] = {}
    for item in sorted(watchlist, key=lambda value: value.rank):
        by_symbol[item.symbol] = UniverseMember(
            symbol=item.symbol,
            name=item.name,
            sources=[UniverseSource.WATCH_PINNED],
            priority=item.rank,
            watch_pinned_rank=item.rank,
            plan_enabled=item.plan_enabled,
            plan_enabled_source="watch_pinned",
            created_at=created_at,
        )

    for position in positions:
        existing = by_symbol.get(position.symbol)
        sources = [UniverseSource.HOLDING]
        watch_rank = None
        priority = 0
        if existing is not None:
            sources.append(UniverseSource.WATCH_PINNED)
            watch_rank = existing.watch_pinned_rank
            priority = existing.priority
        by_symbol[position.symbol] = UniverseMember(
            symbol=position.symbol,
            name=position.name,
            sources=sources,
            priority=priority,
            ledger_updated_at=position.updated_at,
            watch_pinned_rank=watch_rank,
            plan_enabled=True,
            plan_enabled_source="holding",
            created_at=created_at,
        )
    return sorted(by_symbol.values(), key=lambda value: (value.priority, value.symbol))
```

```python
from __future__ import annotations

from datetime import datetime, time
from zoneinfo import ZoneInfo


def plan_valid_until(trading_day: datetime, *, timezone: str) -> datetime:
    tz = ZoneInfo(timezone)
    local_day = trading_day.astimezone(tz).date()
    return datetime.combine(local_day, time(15, 0), tzinfo=tz)


def require_latest_ledger_alignment(
    *,
    latest_ledger_updated_at: datetime | None,
    plan_ledger_updated_at: datetime | None,
) -> str:
    if latest_ledger_updated_at is None or plan_ledger_updated_at is None:
        return "ledger_missing"
    if latest_ledger_updated_at > plan_ledger_updated_at:
        return "ledger_changed"
    return "aligned"
```
